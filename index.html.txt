            const particlesContainer = document.getElementById('particles');

                const particle = document.createElement('div');
                particle.className = 'particle';

                const left = Math.random() * 100;
                const animationDuration = Math.random() * 20 + 10;

                const opacity = Math.random() * 0.5 + 0.2;
                
                particle.style.width = `${size}px`;
                particle.style.height = `${size}px`;
                particle.style.left = `${left}%`;
                particle.style.animationDuration = `${animationDuration}s`;
                particle.style.animationDelay = `${animationDelay}s`;
                particle.style.opacity = opacity;
                
                particlesContainer.appendChild(particle);
            }

        // Animated counter
        function animateCounters() {
            const statElements = [
                { element: document.getElementById('stat1'), target: 2.3, increment: 0.05, duration: 1800, suffix: 'x' },
                { element: document.getElementById('stat2'), target: 600, increment: 10, duration: 1800, suffix: 'L' },
                { element: document.getElementById('stat3'), target: 99.99, increment: 1.5, duration: 1800, suffix: '%' },
                { element: document.getElementById('stat4'), target: 2, increment: 0.05, duration: 1800, suffix: 'M' }
            ];
            
            statElements.forEach(stat => {
                let current = 0;
                const stepTime = Math.abs(Math.floor(stat.duration / (stat.target / stat.increment)));
                const timer = setInterval(() => {
                    current += stat.increment;
                    if (current >= stat.target) {
                        stat.element.textContent = stat.target;
                        clearInterval(timer);
                    } else {
                        stat.element.textContent = current.toFixed(stat.target % 1 !== 0 ? 1 : 0);
                    }
                }, stepTime);
            });
        }
